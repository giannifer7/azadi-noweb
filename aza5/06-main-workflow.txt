# Main workflow coordinating all builds

# .github/workflows/packages-release.yml
name: Build and Release Packages
on:
  push:
    tags:
      - 'v*'

jobs:
  generate:
    runs-on: ubuntu-latest
    steps:
      - uses: actions/checkout@v4
      
      - name: Set up Python
        uses: actions/setup-python@v4
        with:
          python-version: '3.12'
          
      - name: Generate package files
        run: python3 packaging/scripts/generate_packages.py
        
      - name: Upload package files
        uses: actions/upload-artifact@v4
        with:
          name: package-files
          path: packaging/build

  void-glibc:
    needs: generate
    uses: ./.github/workflows/packages-void.yml
    with:
      version: ${{ needs.generate.outputs.version }}
      variant: glibc

  void-musl:
    needs: generate
    uses: ./.github/workflows/packages-void.yml
    with:
      version: ${{ needs.generate.outputs.version }}
      variant: musl

  debian:
    needs: generate
    uses: ./.github/workflows/packages-debian.yml
    with:
      version: ${{ needs.generate.outputs.version }}

  release:
    needs: [generate, void-glibc, void-musl, debian]
    runs-on: ubuntu-latest
    permissions:
      contents: write
    steps:
      - uses: actions/download-artifact@v4
        with:
          path: artifacts

      - name: Create Release
        uses: softprops/action-gh-release@v1
        with:
          files: |
            artifacts/void-glibc-package/*
            artifacts/void-musl-package/*
            artifacts/debian-package/*
          generate_release_notes: true
$$
